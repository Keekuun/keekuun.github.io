function searchMatrix(matrix, target) {
    let m = matrix.length, n = matrix[0].length
    
    function isValid(x, y) {
        return x >= 0 && x < m && y >= 0 && y < n
    }
    
    let row = 0, col = n - 1
    
    while(isValid(row, col) && matrix[row][col] !== target) {
        if(matrix[row][col] < target) {

    }
