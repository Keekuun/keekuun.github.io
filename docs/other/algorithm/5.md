### 1.[有效的括号](https://leetcode-cn.com/problems/valid-parentheses/)
```js
/**
 * @param {string} s
 * @return {boolean}
 */
var isValid = function (s) {
    let map = {
      "}": "{",
      ")": "(",
      "]": "["
    }

    let stack = []

    for (let i = 0; i < s.length; i++) {
      let str = s[i]
      // 遇到右括号
      if (map[str]) {
        // 左括号出栈判断匹配
        if (map[str] !== stack.pop()) {
          return false
        }
      } else {
        // 左括号进栈
        stack.push(str)
      }
    }

    return stack.length === 0
  };
```
