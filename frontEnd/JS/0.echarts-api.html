    trigger: 'item',           // 触发类型，默认数据触发，见下图，可选为：'item' ¦ 'axis'

    transitionDuration : 0.4,         // 动画变换时间，单位s
    backgroundColor: 'rgba(0,0,0,0.7)',  // 提示背景颜色，默认为透明度为0.7的黑色
