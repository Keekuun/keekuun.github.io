    ж•°жҚ®й“ҫи·ҜеұӮ дҝқиҜҒж— е·®й”ҷзҡ„ж•°жҚ®й“ҫи·ҜпјҢдј иҫ“жңүең°еқҖзҡ„её§д»ҘеҸҠй”ҷиҜҜзҡ„жЈҖжөӢеҠҹиғҪ дәӨжҚўжңәгҖҒзҪ‘жЎҘгҖҒзҪ‘еҚЎ SLIPгҖҒCSLIPгҖҒPPPгҖҒARPгҖҒRARPгҖҒMTU
    зү©зҗҶеұӮ дј иҫ“жҜ”зү№жөҒпјҢд»ҘдәҢиҝӣеҲ¶ж•°жҚ®еҪўејҸеңЁзү©зҗҶеӘ’дҪ“дёҠдј иҫ“ж•°жҚ® йӣҶзәҝи·ҜгҖҒдёӯз»§еҷЁ IS02110гҖҒIEEE802гҖҒIEEE802.2
    • TCP/IPеұӮж¬ЎжЁЎеһӢе…ұеҲҶдёәеӣӣеұӮпјҡеә”з”ЁеұӮ->дј иҫ“еұӮ->зҪ‘з»ңеұӮ->ж•°жҚ®й“ҫи·ҜеұӮ

    TCP/IPеҚҸи®®дёҺдҪҺеұӮзҡ„ж•°жҚ®й“ҫи·ҜеұӮе’Ңзү©зҗҶеұӮж— е…і

    зҪ‘з»ңзҹҘиҜҶжўізҗҶ--OSIдёғеұӮзҪ‘з»ңдёҺTCP/IPдә”еұӮзҪ‘з»ңжһ¶жһ„еҸҠдәҢеұӮ/дёүеұӮзҪ‘з»ң (opens new window)

    рҹҢҷ 7.1 HTTP дёҺ HTTPS

    • рҹҢҷ д»Җд№ҲжҳҜ HTTP

    HTTP жҢҮзҡ„жҳҜи¶…ж–Үжң¬дј иҫ“еҚҸи®®пјҲHyperText Transfer ProtocolпјүпјҢе®ғзҡ„зү№зӮ№пјҡ

    • ж— зҠ¶жҖҒпјҡеҚҸи®®еҜ№е®ўжҲ·з«ҜжІЎжңүзҠ¶жҖҒеӯҳеӮЁпјҢеҜ№дәӢзү©еӨ„зҗҶжІЎжңүвҖңи®°еҝҶвҖқиғҪеҠӣпјҢжҜ”еҰӮи®ҝй—®дёҖдёӘзҪ‘з«ҷйңҖиҰҒеҸҚеӨҚиҝӣиЎҢзҷ»еҪ•ж“ҚдҪң
    • ж— иҝһжҺҘпјҡHTTP/1.1д№ӢеүҚпјҢз”ұдәҺж— зҠ¶жҖҒзү№зӮ№пјҢжҜҸж¬ЎиҜ·жұӮйңҖиҰҒйҖҡиҝҮTCPдёүж¬ЎжҸЎжүӢеӣӣж¬ЎжҢҘжүӢпјҢе’ҢжңҚеҠЎеҷЁйҮҚж–°е»әз«ӢиҝһжҺҘгҖӮжҜ”еҰӮжҹҗдёӘе®ўжҲ·жңәеңЁзҹӯж—¶й—ҙеӨҡж¬ЎиҜ·жұӮеҗҢдёҖдёӘиө„жәҗпјҢжңҚеҠЎеҷЁе№¶дёҚиғҪеҢәеҲ«жҳҜеҗҰе·Із»Ҹе“Қеә”иҝҮз”ЁжҲ·зҡ„иҜ·жұӮпјҢжүҖд»ҘжҜҸж¬ЎйңҖиҰҒйҮҚж–°е“Қеә”иҜ·жұӮпјҢйңҖиҰҒиҖ—иҙ№дёҚеҝ…иҰҒзҡ„ж—¶й—ҙе’ҢжөҒйҮҸ
    • еҹәдәҺиҜ·жұӮе’Ңе“Қеә”пјҡеҹәжң¬зҡ„зү№жҖ§пјҢз”ұе®ўжҲ·з«ҜеҸ‘иө·иҜ·жұӮпјҢжңҚеҠЎз«Ҝе“Қеә”
    • з®ҖеҚ•еҝ«йҖҹгҖҒзҒөжҙ»
    • йҖҡдҝЎдҪҝз”ЁжҳҺж–ҮгҖҒиҜ·жұӮе’Ңе“Қеә”дёҚдјҡеҜ№йҖҡдҝЎж–№иҝӣиЎҢзЎ®и®ӨгҖҒж— жі•дҝқжҠӨж•°жҚ®зҡ„е®Ңж•ҙжҖ§
    зүҲжң¬дә§з”ҹж—¶й—ҙеҶ…е®№еҸ‘еұ•зҺ°зҠ¶
    HTTP/0.91991е№ҙдёҚж¶үеҸҠж•°жҚ®еҢ…дј иҫ“пјҢи§„е®ҡе®ўжҲ·з«Ҝе’ҢжңҚеҠЎеҷЁд№Ӣй—ҙйҖҡдҝЎж јејҸпјҢеҸӘиғҪGETиҜ·жұӮжІЎжңүдҪңдёәжӯЈејҸзҡ„ж ҮеҮҶ
    HTTP/1.01996е№ҙдј иҫ“еҶ…е®№ж јејҸдёҚйҷҗеҲ¶пјҢеўһеҠ PUTгҖҒPATCHгҖҒHEADгҖҒ OPTIONSгҖҒDELETEе‘Ҫд»ӨжӯЈејҸдҪңдёәж ҮеҮҶ
    HTTP/1.11997е№ҙжҢҒд№…иҝһжҺҘ(й•ҝиҝһжҺҘ)гҖҒиҠӮзәҰеёҰе®ҪгҖҒHOSTеҹҹгҖҒз®ЎйҒ“жңәеҲ¶гҖҒеҲҶеқ—дј иҫ“зј–з Ғ2015е№ҙеүҚдҪҝз”ЁжңҖе№ҝжіӣ
    HTTP/22015е№ҙеӨҡи·ҜеӨҚз”ЁгҖҒжңҚеҠЎеҷЁжҺЁйҖҒгҖҒеӨҙдҝЎжҒҜеҺӢзј©гҖҒдәҢиҝӣеҲ¶еҚҸи®®зӯүйҖҗжёҗиҰҶзӣ–еёӮеңә

    • рҹҢҷ д»Җд№ҲжҳҜ HTTPS

    HTTPS жҢҮзҡ„жҳҜи¶…ж–Үжң¬дј иҫ“е®үе…ЁеҚҸи®®пјҲHypertext Transfer Protocol SecureпјүпјҢHTTPS е°ұжҳҜе°Ҷ HTTP дёӯзҡ„дј иҫ“еҶ…е®№иҝӣиЎҢдәҶеҠ еҜҶпјҢз„¶еҗҺйҖҡиҝҮеҸҜйқ зҡ„иҝһжҺҘпјҢдј иҫ“еҲ°еҜ№ж–№зҡ„жңәеҷЁдёҠгҖӮеҠ еҜҶзҡ„еҚҸи®®жҳҜ TLS,е…¶еүҚиә«жҳҜ SSLгҖӮ

    • рҹҢҷ дёәд»Җд№ҲиҰҒдҪҝз”Ё HTTPS

    еӣ дёә HTTP жҳҜжҳҺж–Үдј иҫ“зҡ„пјҢеңЁдј иҫ“иҝҮзЁӢдёӯпјҢеңЁдёҖдәӣи·Ҝз”ұиҠӮзӮ№дёӯеҫҲе®№жҳ“иў«еҲ«дәәжҚ•иҺ·пјҢиҜҶеҲ«дҪ зҡ„еҶ…е®№пјҢеҶҚеңЁдј иҫ“зҡ„ж•°жҚ®дёӯж·»еҠ дёҖдәӣе…¶д»–ж•°жҚ®пјҲеҰӮе№ҝе‘ҠпјүгҖӮHTTPS иғҪеҫҲжңүж•Ҳзҡ„и§ЈеҶіиҝҷдёӘзҺ°иұЎпјҢйҖҡиҝҮеҠ еҜҶдј иҫ“еҚідҫҝеңЁи·Ҝз”ұиҠӮзӮ№иў«з¬¬дёүж–№жҚ•иҺ·пјҢе®ғжІЎжңүеҜҶз ҒпјҢзңӢеҲ°зҡ„д№ҹжҳҜдёҖе Ҷд№ұз ҒпјҢHTTPS жӣҙеӨҡзҡ„жҳҜдёәе®үе…ЁиҖғиҷ‘гҖӮ

    • рҹҢҷ HTTPS жҳҜеҰӮдҪ•еҠ еҜҶзҡ„вҖ”вҖ”TLS/SSLеҚҸи®®

    еңЁ TLS/SSL дёӯпјҢдё»иҰҒе®һзҺ°зҡ„е°ұжҳҜеҠ еҜҶзҡ„иҝҮзЁӢгҖӮеҠ еҜҶиҝҮзЁӢдёӯз”ЁеҲ°дәҶгҖҢйқһеҜ№з§°еҠ еҜҶгҖҚе’ҢгҖҢеҜ№з§°еҠ еҜҶгҖҚдҝ©з§ҚеҠ еҜҶж–№ејҸгҖӮ

    еҜ№з§°еҠ еҜҶжҖқи·ҜжҜ”иҫғз®ҖеҚ•пјҢе°ұжҳҜеӨ§е®¶зәҰе®ҡдёҖдёӘеҜҶз ҒиҝӣиЎҢеҠ еҜҶпјҢз„¶еҗҺз”ЁеҗҢдёҖдёӘеҜҶз ҒеңЁиҝӣиЎҢи§ЈеҜҶпјҢйҖ»иҫ‘жҜ”иҫғз®ҖеҚ•гҖӮпјҲеҜ№з§°еҠ еҜҶеҜҶз®ҖеҚ•й«ҳж•ҲпјҢжҖ§иғҪй«ҳпјҢдёҚе®үе…Ёпјү

    йқһеҜ№з§°еҠ еҜҶж–№жі•дёҖејҖе§Ӣдјҡз”ҹжҲҗдёҖеҜ№еҜҶй’ҘпјҢд№ҹе°ұжҳҜдҝ©дёӘеҜҶз ҒгҖӮе°ұжҳҜиҜҙиҝҷдёӘеҠ еҜҶи§ЈеҜҶиҝҮзЁӢдјҡз”Ёдҝ©дёӘеҜҶй’ҘжқҘе®ҢжҲҗгҖӮеҰӮжһңдҪ з”Ёз¬¬дёҖжҠҠй’ҘеҢҷеҠ еҜҶпјҢйӮЈд№ҲдҪ еҸӘиғҪз”Ёз¬¬дәҢжҠҠй’ҘеҢҷжүҚиғҪи§ЈејҖгҖӮдҪ з”Ёз¬¬дәҢжҠҠй’ҘеҢҷеҠ еҜҶпјҢеҸӘиғҪз”Ёз¬¬дёҖжҠҠй’ҘеҢҷи§ЈејҖпјҢиҝҷжҳҜйқһеҜ№з§°еҠ еҜҶзҡ„иЎЁзҺ°гҖӮпјҲеҰӮRSAеҠ еҜҶз®—жі•пјүпјҲйқһеҜ№з§°еҠ еҜҶе®үе…ЁпјҢжҖ§иғҪдҪҺпјү

    TLS/SSLз»“еҗҲйқһеҜ№з§°еҠ еҜҶе®үе…Ёе’ҢеҜ№з§°еҠ еҜҶеҜҶз®ҖеҚ•й«ҳж•Ҳзҡ„дјҳзӮ№пјҢе…ҲдҪҝз”ЁйқһеҜ№з§°еҠ еҜҶз®—жі•е°ҶжҹҗдёҖеҜҶж–ҮеҠ еҜҶпјҢд№ӢеҗҺзҡ„жүҖжңүж•°жҚ®йҮҮз”ЁиҜҘеҜҶж–ҮиҝӣиЎҢеҜ№з§°еҠ еҜҶгҖӮиҝҷж ·пјҢж—ўдҝқиҜҒдәҶжҖ§иғҪпјҢеҸҲдҝқиҜҒдәҶе®үе…ЁжҖ§гҖӮ

    • httpsзҪ‘з«ҷжөҸи§Ҳж—¶зҡ„еҠ и§ЈеҜҶе…·дҪ“иҝҮзЁӢ

    еңЁйҮҮз”ЁйқһеҜ№з§°еҠ еҜҶз®—жі•з”ҹжҲҗдҝ©дёӘй’ҘеҢҷд№ӢеҗҺпјҢзҪ‘з«ҷйңҖиҰҒе…¬еёғеҮәеҺ»е…¶дёӯдёҖжҠҠй’ҘеҢҷпјҢжҲ‘д»¬жҠҠе…¬еёғеҮәеҺ»зҡ„еҜҶз Ғз§°д№Ӣдёәе…¬й’ҘпјҢиҮӘе·ұз•ҷзқҖзҡ„йӮЈдёҖжҠҠй’ҘеҢҷз§°д№Ӣдёәз§Ғй’ҘгҖӮжҲ‘д»¬еҗ‘е…Ёдё–з•Ңе№ҝж’ӯжҲ‘д»¬зҪ‘з«ҷе…¬й’ҘжҳҜд»Җд№ҲпјҢеӨ§е®¶йғҪеҸҜд»ҘйҖҡиҝҮиҝҷдёӘе…¬й’ҘеҜ№иҰҒдј иҫ“зҡ„ж•°жҚ®иҝӣиЎҢеҠ еҜҶпјҢз„¶еҗҺжҲ‘д»¬зҪ‘з«ҷеҶ…йғЁдјҡжңүдёҖдёӘз§Ғй’ҘпјҢеҸҜд»ҘйҖҡиҝҮз§Ғй’Ҙи§ЈеҜҶе…¬й’ҘеҠ еҜҶзҡ„ж•°жҚ®гҖӮ

    жҖҺд№Ҳе°Ҷе…¬й’ҘдәӨз»ҷдҪ е‘ў

    е°ұжҳҜжөҸи§ҲеҷЁе’ҢжңҚеҠЎеҷЁе»әз«Ӣиө·дәҶиҝһжҺҘд№ӢеҗҺпјҢжөҸи§ҲеҷЁдјҡе…ҲеҸ‘йҖҒдёҖдёӘдҝЎжҒҜпјҢе°ұжҳҜиҮӘе·ұзҡ„е…¬й’ҘпјҲе…¬ејҖзҡ„еҜҶз ҒпјүпјҢз„¶еҗҺжөҸи§ҲеҷЁжӢҝеҲ°иҝҷдёӘеҜҶз ҒеҜ№ж•°жҚ®иҝӣиЎҢеҠ еҜҶпјҢеҠ еҜҶд№ӢеҗҺдј иҫ“з»ҷжңҚеҠЎеҷЁпјҢжңҚеҠЎеҷЁж”¶еҲ°еҜҶж–Үд№ӢеҗҺз”ЁиҮӘе·ұзҡ„з§Ғй’Ҙи§ЈеҜҶж•°жҚ®пјҢиҺ·еҫ—зңҹе®һж•°жҚ®пјҢе®ҢжҲҗдәҶеҠ еҜҶдј иҫ“зҡ„жғіжі•гҖӮ

    иҝҷз§Қж–№ејҸпјҢдҪҶд№ҹжңүе®үе…ЁйЈҺйҷ©пјҢеӣ дёәеңЁдј иҫ“е…¬й’Ҙзҡ„иҝҮзЁӢдёӯпјҢз¬¬дёүж–№жҲӘеҸ–дәҶе…¬й’ҘпјҢз„¶еҗҺйҡҸдҫҝз»ҷдҪ еҸ‘дёҖдёӘеҜҶз ҒпјҢдҪ дјҡд»ҘдёәиҝҷдёӘеҜҶз Ғе°ұжҳҜжңҚеҠЎеҷЁеҸ‘жқҘзҡ„пјҢжңҚеҠЎеҷЁд№ҹдёҚзҹҘйҒ“еҜҶз Ғиў«жҲӘиҺ·дәҶгҖӮдәҺжҳҜе°ұжңүзҡ„CAиҜҒд№ҰгҖӮ

    • рҹҢҷ д»Җд№ҲжҳҜCA

    ж•°еӯ—иҜҒд№Ұи®ӨиҜҒжңәжһ„пјҲCertificate AuthorityпјҢзј©еҶҷдёәCAпјүгҖӮ

    return new Promise((resolve, reject) => {
        let script = document.createElement('script');
        whidow[callback] = function(data) {
            resolve(data);
            document.body.removeChild(script);
        }
        
        params = {...params, callback} // wd=hello&callback=search
        
        let arr = [];
        for(let key in params) {

        }

        document.body.appendChild(script);
    })

    рҹҢҷ 12.2 еҜ№дәҺ useEffect е’Ң useLayoutEffect е“ӘдёҖдёӘдёҺ componentDidMountпјҢcomponentDidUpdate зҡ„жҳҜзӯүд»·зҡ„пјҹ

    useLayoutEffectпјҢеӣ дёәд»Һжәҗз Ғдёӯи°ғз”Ёзҡ„дҪҚзҪ®жқҘзңӢпјҢuseLayoutEffectзҡ„ create еҮҪж•°зҡ„и°ғз”ЁдҪҚзҪ®гҖҒж—¶жңәйғҪе’Ң componentDidMountпјҢcomponentDidUpdate дёҖиҮҙпјҢдё”йғҪжҳҜиў« React еҗҢжӯҘи°ғз”ЁпјҢйғҪдјҡйҳ»еЎһжөҸи§ҲеҷЁжёІжҹ“гҖӮ

    рҹҢҷ 12.3 useEffect е’Ң useLayoutEffect е“ӘдёҖдёӘдёҺ componentWillUnmount зҡ„жҳҜзӯүд»·зҡ„пјҹ

function compose(...fns) {
  if (fns.length === 0) {
    return a => a;
  }
  if (fns.length === 1) {
    return fns[0];
  }

  var fn = fns.reduce((a, b) => {
    return (...args) => {
      return a(b(...args));
    };
  });
  return fn;
}

function applyMiddleware(...middlewares) {
  return createStore => (...args) => {

function throttle(fn, wait, {leading = true, trailing = true} = {}) {
  let timeout
  let previous = 0
  return function(...args) {
    let now = new Date().getTime()
    if (!previous && leading === false) previous = now
    let remaining = wait - (now - previous)
    if (remaining <= 0 || remaining > wait) {
      if (timeout) {
        clearTimeout(timeout)
        timeout = null
      }
      previous = now
      fn.apply(context, args)
    } else if (!timeout && trailing !== false) {
      // еҰӮжһңжңүеү©дҪҷж—¶й—ҙдҪҶе®ҡж—¶еҷЁдёҚеӯҳеңЁпјҢдё”trailingдёҚдёәfalseпјҢеҲҷи®ҫзҪ®е®ҡж—¶еҷЁ

        previous = leading === false ? 0 : new Date().getTime()
        timeout = null
        fn.apply(this, args)
      }, remaining)
    }
  }
